```javascript{.line-numbers}
import { router } from '@kit.ArkUI';         // 导入ArkUI框架的路由功能模块
import { BusinessError } from '@kit.BasicServicesKit'; // 导入基础服务错误类型定义

@Entry                                      // 装饰器：标记当前组件为应用入口组件
@Component                                  // 装饰器：声明这是一个自定义组件
struct Index {                              // 定义名为Index的结构体组件
  @State message: string = 'Hello World'    // 声明响应式状态变量，初始值为'Hello World'

  build() {                                 // 实现build方法，用于构建组件UI
    Row() {                                 // 创建横向布局容器（行布局）
      Column() {                            // 创建纵向布局容器（列布局），作为Row的子组件
        Text(this.message)                  // 创建文本组件，显示message变量内容
          .fontSize(50)                    // 设置文本字体大小（链式调用）
          .fontWeight(FontWeight.Bold)      // 设置字体加粗样式（链式调用）

        Button() {                          // 创建按钮组件
          Text('跳转')                      // 按钮内部文本组件
            .fontSize(30)                  // 设置按钮文字大小
            .fontWeight(FontWeight.Bold)    // 设置按钮文字加粗样式
        }                                   // 结束Button组件的定义
        .type(ButtonType.Capsule)           // 设置按钮样式为胶囊型
        .margin({                           // 设置按钮外边距
          top: 20                           // 顶部外边距20单位
        })                                  // 结束margin属性设置
        .backgroundColor('#0D9FFB')         // 设置按钮背景颜色
        .width('40%')                       // 设置按钮宽度为父容器的40%
        .height('5%')                      // 设置按钮高度为父容器的5%

        .onClick(() => {                    // 绑定按钮点击事件处理函数
          console.info(`Succeeded in clicking the 'Next' button.`) // 打印点击日志

          // 使用路由模块进行页面跳转
          router.pushUrl({ url: 'pages/Second' }) // 调用路由跳转到pages/Second页面
            .then(() => {                   // Promise成功回调
              console.info('Succeeded in jumping to the second page.')  // 当跳转成功时，执行 then 方法中的回调函数，在控制台输出跳转成功的提示信息
            })                              // 结束then回调
            .catch((err: BusinessError) => { // Promise失败回调
              console.error(`Failed to jump. Code: ${err.code}, message: ${err.message}`)   // 当跳转过程中出现错误时，执行 catch 方法中的回调函数，在控制台输出跳转失败的提示信息，包含错误代码和错误消息
            })                             // 结束catch回调
        })                                  // 结束onClick事件绑定
      }                                     // 结束Column组件的定义
      .width('100%')                        // 设置Column宽度占满父容器
    }                                       // 结束Row组件的定义
    .height('100%')                         // 设置Row高度占满父容器
  }                                         // 结束build方法的定义
}                                           // 结束Index组件的结构体定义


```

## pages2 
```javascript{.line-numbers}
import { router } from '@kit.ArkUI';         // 导入ArkUI路由模块
import { BusinessError } from '@kit.BasicServicesKit'; // 导入错误类型定义

@Entry                                      // 装饰器：标记为页面入口组件
@Component                                  // 装饰器：声明为可复用组件
struct Second {                             // 定义Second页面结构体
  @State message: string = '第二页'         // 声明响应式状态变量，初始值为'第二页'

  build() {                                 // UI构建方法
    Row() {                                 // 创建横向布局容器
      Column() {                            // 创建纵向布局容器作为子组件
        Text(this.message)                  // 显示页面标题文本
          .fontSize(50)                    // 设置字体大小
          .fontWeight(FontWeight.Bold)      // 设置字体加粗样式

        Button() {                          // 创建返回按钮
          Text('返回')                      // 按钮文本内容
            .fontSize(30)                   // 设置按钮文字大小
            .fontWeight(FontWeight.Bold)     // 设置文字加粗样式
        }                                   // 结束Button组件定义
        .type(ButtonType.Capsule)           // 设置按钮样式为胶囊型
        .margin({                           // 设置外边距
          top: 20                           // 顶部间距20单位
        })                                  // 结束margin设置
        .backgroundColor('#0D9FFB')         // 设置按钮背景颜色
        .width('40%')                       // 设置按钮宽度
        .height('5%')                       // 设置按钮高度

        .onClick(() => {                    // 绑定按钮点击事件
          console.info(`Succeeded in clicking the 'Back' button.`) // 打印日志
          try {                             // 异常处理try块开始
            router.back()                   // 调用路由返回功能
            console.info('Succeeded in returning to the first page.') // 成功日志
          } catch (err) {                   // 捕获异常
            let code = (err as BusinessError).code;    // 类型断言并获取错误码
            let message = (err as BusinessError).message; // 获取错误信息
            console.error(`返回失败，错误码：${code}，信息：${message}`) // 错误日志
          }                                 // 结束catch块
        })                                  // 结束onClick事件绑定
      }                                     // 结束Column组件定义
      .width('100%')                       // 设置Column宽度占满父容器
    }                                       // 结束Row组件定义
    .height('100%')                         // 设置Row高度占满屏幕
  }                                         // 结束build方法定义
}                                           // 结束Second组件结构体
```
