## page1
```javascript{.line-numbers}
// Index.ets
// 导入页面路由模块
import { router } from '@kit.ArkUI'; // 导入ArkUI框架中的路由模块，用于页面导航
// 配置页面路由
import { BusinessError } from '@kit.BasicServicesKit'; // 导入基础服务模块中的错误处理类，用于捕获和处理错误
// 接入调试功能

@Entry // 标记当前组件为应用的入口组件
@Component // 标记当前结构体为一个组件
struct Index { // 定义一个名为Index的组件结构体
  @State message: string = 'Hello World'; // 使用@State装饰器定义一个状态变量message，初始值为'Hello World'

  build() { // 组件的UI构建方法，用于定义组件的布局和内容
    // 自定义声明式节点
    Row() { // 创建一个行布局容器
      Column() { // 创建一个列布局容器
        Text(this.message) // 创建一个文本组件，显示message的内容
          .fontSize(50) // 设置文本的字体大小为50
          .fontWeight(FontWeight.Bold) // 设置文本的字体加粗
        // 添加按钮，以响应用户点击
        Button() { // 创建一个按钮组件
          Text('Next') // 按钮内显示文本'Next'
            .fontSize(30) // 设置按钮文本的字体大小为30
            .fontWeight(FontWeight.Bold) // 设置按钮文本的字体加粗
        } //表示 Button 组件内部元素定义的结束
        .type(ButtonType.Capsule) // 设置按钮的类型为胶囊形状
        .margin({ // 设置按钮的外边距
          top: 20 // 上边距为20
        }) // 表示 margin 方法传入的对象字面量结束
        .backgroundColor('#0D9FFB') // 设置按钮的背景颜色为#0D9FFB
        .width('40%') // 设置按钮的宽度为父容器宽度的40%
        .height('5%') // 设置按钮的高度为父容器高度的5%
        // 跳转按钮绑定onClick事件，点击时跳转到第二页
        .onClick(() => { // 设置按钮的点击事件处理函数
          console.info(`Succeeded in clicking the 'Next' button.`) // 打印日志，表示按钮点击成功
          // 跳转到第二页
          router.pushUrl({ url: 'pages/second' }) // 调用路由的pushUrl方法，跳转到指定页面（pages/second）
            .then(() => { // 跳转成功时的回调
              console.info('Succeeded in jumping to the second page.') // 打印日志，表示跳转成功
            })
            .catch((err: BusinessError) => { // 跳转失败时的回调
              console.error(`Failed to jump to the second page. Code is ${err.code}, message is ${err.message}`) // 打印错误日志
            }) //表示catch组件设置结束
        }) // 表示onClick组件设置结束
      }// 表示 Column 容器内子组件定义的结束
      .width('100%') // 设置列布局的宽度为父容器的100%
    }
    .height('100%') // 设置行布局的高度为父容器的100%
  }  // build 方法结束
}  // 结构体定义结束

```

## page2
```javascript{.line-numbers}
// second.ets
// 导入页面路由模块
import { router } from '@kit.ArkUI';  // 导入ArkUI框架中的路由模块
import { BusinessError } from '@kit.BasicServicesKit';  // 导入基础服务模块中的错误处理类

@Entry  // 标记当前组件为入口组件
@Component  // 标记当前结构体为组件
struct second {  // 定义名为second的组件结构体
  @State message: string = 'Hi there';  // 定义状态变量message，初始值为'Hi there'

  build() {  // 定义组件的UI构建方法
    Row() {  // 创建一个行布局
      Column() {  // 创建一个列布局
        Text(this.message)  // 创建一个文本组件，显示message的内容
          .fontSize(50)  // 设置文本字体大小为50
          .fontWeight(FontWeight.Bold)  // 设置文本字体加粗
        Button() {  // 创建一个按钮组件
          Text('Back')  // 按钮内显示文本'Back'
            .fontSize(30)  // 设置按钮文本字体大小为30
            .fontWeight(FontWeight.Bold)  // 设置按钮文本字体加粗
        } // 按钮组件内容定义结束
        .type(ButtonType.Capsule)  // 设置按钮类型为胶囊形状
        .margin({  // 设置按钮的外边距
          top: 20  // 上边距为20
        }) // 边距对象结束
        .backgroundColor('#0D9FFB')  // 设置按钮背景颜色为#0D9FFB
        .width('40%')  // 设置按钮宽度为父容器的40%
        .height('5%')  // 设置按钮高度为父容器的5%
        // 返回按钮绑定onClick事件，点击按钮时返回到第一页
        .onClick(() => {  // 设置按钮的点击事件处理函数
          console.info(`Succeeded in clicking the 'Back' button.`)  // 打印日志，表示按钮点击成功
          try { // 异常处理开始
            // 返回第一页
            router.back()  // 调用路由的back方法，返回上一页
            console.info('Succeeded in returning to the first page.')  // 打印日志，表示返回成功
          } catch (err) {  // 捕获异常
            let code = (err as BusinessError).code;  // 获取错误代码
            let message = (err as BusinessError).message;  // 获取错误信息
            console.error(`Failed to return to the first page. Code is ${code}, message is ${message}`)  // 打印错误日志
          } // try-catch 块结束
        }) // 点击事件回调定义结束
      } // Column 组件内容结束
      .width('100%')  // 设置列布局的宽度为100%
    } // Row 组件内容结束
    .height('100%')  // 设置行布局的高度为100%
  } // build 方法结束
} // 结构体定义结束
